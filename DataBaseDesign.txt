User Collection :
    - _id : ObjectId()
    - email : String
    - password : String (stored in hashed format)
    - name : String
    - rollNumber : String
    - role : Enum ("student" / "faculty")
    - course : String 
    - year : Number
    - isAdmin : Boolean
    - department : String
    - subject : [String]
    - subcriptionDetails : [SubcriptionDataSchema]
